def deuda(entidad)
  if entidad == agency
    agencia_deuda
  elsif operadora == entidad
    operadora_deuda
  else
    raise "Se ha pedido la deuda de #{entidad} en la reserva: #{self}, pero esta entidad no tiene nada que ver"
  end
end
